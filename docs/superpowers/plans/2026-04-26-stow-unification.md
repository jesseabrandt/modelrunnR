# Unified `stow()` Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Collapse `ingest()` into `stow()` via type dispatch on a tagged `mr_file` value, add `shape = "versioned"` for in-memory frames, and deprecate `ingest()` to a thin shim — closing the verb-mismatch and path-only friction points together.

**Architecture:** `stow()` already type-dispatches (lazy tbl → append, data frame → append, anything else → versioned artifact). This change adds (a) a new value class `mr_file` whose dispatch branch routes through the existing file-ingest internals, and (b) a `shape = "versioned"` opt-in for tabular values that routes them through the existing `.mr_stow_table()` versioned-shape internal. `ingest()` becomes `.Deprecated()` + delegate. No schema change. No new `Imports`.

**Tech Stack:** R / S3 dispatch (single-class tagged value), DuckDB via existing `.mr_*` internals, testthat 3, roxygen2.

**Spec:** `docs/superpowers/specs/2026-04-26-stow-unification-design.md`

```r
# R/mr_file.R

#' Tag a file path as a modelrunnR file source
#'
#' Wraps a path in a class (`mr_file`) that [stow()] dispatches on, so
#' a file source flows through the same `stow()` verb as any other
#' value. Validation is lazy — the file does not need to exist at
#' construction time; existence is checked when the value is handed to
#' `stow()`.
#'
#' @param path Length-1 non-empty character: a path to a `.csv`,
#'   `.tsv`, or `.parquet` file.
#'
#' @return A length-1 character vector of class
#'   `c("mr_file", "character")`.
#'
#' @examples
#' \dontrun{
#' stow(mr_file("data/training.parquet"), "training")
#' }
#' @export
mr_file <- function(path) {
  if (!is.character(path) || length(path) != 1L ||
      is.na(path) || !nzchar(path)) {
    stop("mr_file(): `path` must be a length-1 non-empty character.",
         call. = FALSE)
  }
  structure(path, class = c("mr_file", "character"))
}
```

```r
# Implementation for stowing a file source. Stages the file into a
# DuckDB temp table via .mr_ingest_file_to_table(), hashes the staged
# table, and atomically promotes it to the canonical physical name (or
# drops it if a version with the same content already exists). Records
# source_uri / source_hash on the _mr_versions row.
#
# Called from:
#   - stow.mr_file() (the new dispatch branch in R/stow.R)
#   - ingest() (deprecation shim in R/ingest.R)
.mr_stow_file <- function(name, path) {
  .mr_validate_name(name, context = "stow")
  stopifnot(
    is.character(path),
    length(path) == 1L,
    nzchar(path)
  )
  if (!file.exists(path)) {
    stop(sprintf("stow(): file not found: %s", path), call. = FALSE)
  }

  con <- .mr_get_connection()

  staging <- paste0(
    "_mr_tmp_ingest_",
    paste(sample(c(0:9, letters), 10, replace = TRUE), collapse = "")
  )
  .mr_ingest_file_to_table(con, path, staging)
  staging_alive <- TRUE
  on.exit(
    if (staging_alive) try(.mr_drop_table(con, staging), silent = TRUE),
    add = TRUE
  )

  content_hash <- .mr_hash_duckdb_table(con, staging)
  physical_name <- .mr_physical_name(name, content_hash)
  src_hash      <- .mr_file_hash(path)
  src_uri       <- normalizePath(path, mustWork = TRUE)

  existing <- .mr_get_version_row(con, name, content_hash)
  now <- Sys.time()

  DBI::dbBegin(con)
  tryCatch({
    if (nrow(existing) == 0L) {
      .mr_execute(
        con,
        sprintf(
          "ALTER TABLE %s RENAME TO %s",
          .mr_quote_ident(staging), .mr_quote_ident(physical_name)
        )
      )
      DBI::dbExecute(
        con,
        "INSERT INTO _mr_versions
           (logical_name, content_hash, physical_name, kind,
            first_seen, last_seen, size_bytes, storage_location,
            source_uri, source_hash)
         VALUES (?, ?, ?, 'table', ?, ?, ?, NULL, ?, ?)",
        params = list(name, content_hash, physical_name, now, now,
                      0, src_uri, src_hash)
      )
      .mr_refresh_latest_view(con, name)
    } else {
      DBI::dbExecute(
        con,
        "UPDATE _mr_versions
           SET last_seen = ?, source_uri = ?, source_hash = ?
         WHERE logical_name = ? AND content_hash = ?",
        params = list(now, src_uri, src_hash, name, content_hash)
      )
    }
    DBI::dbCommit(con)
    staging_alive <<- FALSE
  }, error = function(e) {
    DBI::dbRollback(con)
    stop(e)
  })

  .mr_record_write(name, content_hash)
  .mr_maybe_record_interactive_write(name, content_hash)
  .mr_maybe_warn_version_count(con, name)

  invisible(dplyr::tbl(con, physical_name))
}
```

```r
      "stow() is value-first as of this version: stow(value, name). ",
      "Did you mean `stow(<value>, \"", value, "\")` ?",
      call. = FALSE
    )

  if (!missing(name) && is.character(value) && length(value) == 1L &&
      !is.character(name)) {
    stop(
      "stow() is value-first as of this version: stow(value, name). ",
      "Did you mean `stow(<value>, \"", value, "\")` ?",
      call. = FALSE
    )
  }
  .mr_validate_name(name, context = "stow")

  # Validate `shape` (used in Task 6 to route frame-as-versioned).
  if (!is.null(shape)) {
    if (!is.character(shape) || length(shape) != 1L ||
        !shape %in% c("versioned", "append")) {
      stop(
        'stow(): shape must be NULL, "versioned", or "append".',
        call. = FALSE

        "shape argument.",
        call. = FALSE

        sprintf(
          "stow(): shape is only meaningful for data frames and lazy tbls; got %s.",
          paste(class(value), collapse = "/")
        ),
        call. = FALSE
      )
    }
  }
```
